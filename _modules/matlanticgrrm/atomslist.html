[ドキュメント]def translate(atoms_list,displacement):
    """並行移動する
    
    Parameters:
    
    atoms_list: list of ase.Atoms
        ase.Atomsを要素とするリスト.
    displacement: list
        1*3のリスト[x(float),y(float),z(float)]の場合，全ての構造を同じ変位で並行移動させる．
        [1*3]*nの2Dリストで与えた場合,各要素ごとに異なる変位で並行移動させることができる．
        
    Note:
        破壊的メソッドなので注意!!.特にnote-book形式で実行する場合は慎重に行なう
    """
    if type(displacement[0]) == list:
        """各要素に異なる並行移動を行なう"""
        for atoms,d in zip(atoms_list,displacement):
            if atoms:
                atoms.translate(d)
    else:
        for atoms in atoms_list:
            if atoms:
                atoms.translate(displacement)

[ドキュメント]def compare(atoms_list1,atoms_list2,target0=None,target1=[],target2=[],mult=1,**kwargs):
    """2つのatoms_listが同じ構造か調べる
    
    atoms_list1とatoms_list2の1番目の要素同士,2番目の要素同士...で比較する
    
    Parameters:
    
    atoms_list1: ase.Atoms or list of ase.Atoms
        ase.Atomsを要素とするリスト.またはase.Atoms
    atoms_list2: ase.Atoms or list of ase.Atoms
        ase.Atomsを要素とするリスト.またはase.Atoms
        
    Return: bool or list of bool:
        構造の同じものはTrueを返す
        atoms_listをリストで与えた場合，boolとNoneを格納したリストを返す
        atoms_list1,atoms_list2のどちらか一方の要素がNoneの場合はNoneの要素が格納される.
    """
    if len(atoms_list1) != len(atoms_list2):
        raise Exception("atoms_list1とatoms_list2の要素数が一致しません")
    smile1 = get_smiles(atoms_list1,target0,target1,target2,mult,**kwargs)
    smile2 = get_smiles(atoms_list2,target0,target1,target2,mult,**kwargs)
    if type(atoms_list1) == list:
        bool_list = []
        for a,b in zip(smile1,smile2):
            if a is None or b is None:
                bool_list.append(None)
            elif a == b:
                bool_list.append(True)
            else:
                bool_list.append(False)
        return bool_list
    elif type(atoms_list1) == ase.Atoms:
        if smile1 is None or smile2 is None:
            return False
        elif smile1 == smile2:
            return True
        else:
            return False

    """atoms_listの内,同じ構造を探す．
    
    Parameters:
    
    target_atoms: ase.Atoms or list of ase.Atoms
        ase.Atomsを要素とするリスト.またはase.Atoms, 検索対象
    range_atoms: ase.Atoms or list of ase.Atoms

        target_atomsをリストで与えた場合は2次元リストを返す
    """
    target_smiless = get_smiles(target_atoms,target0,target1,target2,mult,**kwargs) # ->list
    range_smiless = get_smiles(range_atoms,target0,target1,target2,mult,**kwargs) # ->list
    result_list = []
    for target_smiles in target_smiless:
        result = []
        for i, range_siles in enumerate(range_smiless):
            if target_smiles is None or range_siles is None:
                pass
            elif target_smiles == range_siles:
                result.append(i)
        result_list.append(result)
    
    if type(target_atoms) == ase.Atoms:
        return result_list[0]
    else:
        return result_list

                

[ドキュメント]def grouping(atoms_list,target0=None,target1=[],target2=[],output_type=1,mult=1,**kwargs):
    """atoms_listの内,同じ構造をグループする.
    
    Parameters:
    
    atoms_list: list of ase.Atoms
        ase.Atomsを要素とするリスト.
    target0: list of int
        target0, target1, target2に指定した原子同士の結合を見る
    target1: list of int
        target0に指定した原子との結合を見る
    target2: list of int
        target0, target2に指定した原子同士の結合を見る
    output_type: 1 or 2
        戻り値の形式
    mult: float
        カットオフの係数
    kwargs: 
        元素の名前と共有結合半径を指定(例:Al=1.2), カットオフ値が上書きされる.
               
    Note:
        output_type=1の場合,atoms_listと同じ要素数の1Dリストを返す.要素には数値が入っており,同じ構造は同じ数値が入る.
        つまり,数値=グループ名.
        output_type=2の場合,2Dリストを返す.各要素(リスト)には同じ構造と判断されたatoms_listのindex番号が入っている.
        つまり,数値=atoms_listのindex番号であり,要素数=グループの数である.
        
    Note:
        RDkitを用いて,カノニカルSMILESを比較することで同じ構造かどうかを判定している. 
        結合しているかどうかの判断はASEを用いている.
        カノニカルsmileで比較しているので,1-シクロヘキサンと2-シクロヘキサンは同一であると判断される．
        稀に異なる構造でも同じSMILESを返すことがあるので注意
    """
    smiles_list = get_smiles(atoms_list,target0,target1,target2,mult,**kwargs)
            
    unique_smile = sorted(set(smiles_list), key=smiles_list.index)
    gtoup_dict = dict(zip(unique_smile,[i for i in range(len(unique_smile))])) # gtoup_dict:{SMILES_a:0,SMILES_b:1,...}
    if output_type == 1:    
        group_name_list = [gtoup_dict[smile] for smile in smiles_list]
        return group_name_list
    elif output_type == 2:
        gtoup_n = max(gtoup_dict.values())+1
        group_list = [[] for i in range(gtoup_n)]
        for i,smiles in enumerate(smiles_list):
            group_list[gtoup_dict[smiles]].append(i)
        return group_list

            

[ドキュメント]def get_smiles(atoms_list,target0=None,target1=[],target2=[],mult=1,**kwargs):
    """カノニカルSMILESを出力する.
    
    Parameters:
    
    atoms_list: ase.Atoms or list of ase.Atoms
        ase.Atomsまたは,そのリスト
    target0: list of int
        target0, target1, target2に指定した原子同士の結合を見る,デフォルトでは全ての原子
    target1: list of int
        target0に指定した原子との結合を見る
    target2: list of int
        target0, target2に指定した原子同士の結合を見る
    mult: float
        カットオフの係数(数字が大きいと離れていても結合しているとみなされる)
    kwargs: 
        元素の共有結合半径を上書きする(例: Al=1.1)
        
    Return:
        カノニカルSMILESのリスト
    """
    ####targetの調整###############
    def renumbering():
        """target3があった場合にtarget0,1,2の番号を振り直す"""
        new_target0 = [] 
        new_target1 = [] 
        new_target2 = []
        n = 0
        for i in range(n_atoms):
            if not i in target3:
                if i in target0:
                    new_target0.append(n)
                elif i in target1:
                    new_target1.append(n)
                elif i in target2:
                    new_target2.append(n)
                n += 1
        return new_target0,new_target1,new_target2
    
    if type(atoms_list) == list:
        for atoms in atoms_list:
            if atoms:
                n_atoms = len(atoms)
                break
    elif type(atoms_list) == ase.Atoms:
        n_atoms = len(atoms_list)
            
    if target0 is None:
        target0 = [i for i in range(n_atoms)]

    target3 = [i for i in range(n_atoms) if not i in target0+target1+target2] # どれとも結合しない原子

    target0,target1,target2 = renumbering()
    
    
    #####RDkitのMolオブジェクトのリストを作成############
    if type(atoms_list) == ase.Atoms:
        """もしatoms_listがリストでなく単一のase.Atomsで与えられていたら"""
        atoms_list = [atoms_list]
    mols = []
    for atoms in atoms_list:
        if atoms:
            atoms = atoms.copy()
            del atoms[target3]
            m = Chem.MolFromSmiles('') 
            mol = Chem.RWMol(m) #空のmolオブジェクトを作成
            for sybmols in atoms.get_chemical_symbols():
                """原子を追加するこの時点では結合はない"""
                mol.AddAtom(Chem.Atom(sybmols))
                
            cutoff = natural_cutoffs(atoms, mult=mult,**kwargs) # cutoff半径(結合の許容度)を設定，multは係数，maltが大きいと結合しているとみなされる
            nl = build_neighbor_list(atoms, cutoff)
            ana = Analysis(atoms, nl=nl)
            bonds = ana.unique_bonds
            for a_idx,bond in enumerate(bonds[0]):
                """結合を追加する"""
                for b_idx in bond:
                    if a_idx in target0 or b_idx in target0:
                        mol.AddBond(int(a_idx),int(b_idx),Chem.BondType.SINGLE)
                    elif a_idx in target2 and b_idx in target2:
                        mol.AddBond(int(a_idx),int(b_idx),Chem.BondType.SINGLE)
            mols.append(mol)
        else:
            mols.append(None)
            
    smiles_list = [Chem.MolToSmiles(mol) if mol else None for mol in mols]
    return smiles_list # ase.Atomsで与えたとしても返り値はリスト
